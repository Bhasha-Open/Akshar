Hinglish normalization utilities
--------------------------------

.. code-block:: python

   from akshar.normalize import remove_elongations, roman_phonetic_signature

   for w in ["heyyy", "yaaaaar", "niceeee", "bohoooot"]:
       print(w, "‚Üí", remove_elongations(w))

   print("\nPhonetic signatures:")
   for group in [["nahi","nahii","nahee"], ["yaar","yar"], ["achha","acha","accha"]]:
       print(group, "‚Üí", [roman_phonetic_signature(w) for w in group])
